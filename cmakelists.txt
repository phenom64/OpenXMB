cmake_minimum_required(VERSION 3.19)

project(OpenXMB LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

find_package(Qt6 REQUIRED COMPONENTS Core Gui Quick Multimedia ShaderTools)

add_executable(XMS
  src/main.cpp
  src/WaveItem.h
  src/WaveItem.cpp
  src/XmbColorScheme.h
  src/XmbColorScheme.cpp
)

# QML, audio, fonts
qt6_add_resources(XMS resources.qrc)

# Bake Vulkan-friendly shaders to QSB (SPIR-V is embedded for Vulkan)
qt6_add_shaders(XMS xms_shaders
  PREFIX "/shaders"
  FILES
    shaders/WaveRhi.vert
    shaders/WaveRhi.frag
)

target_link_libraries(XMS PRIVATE
  Qt6::Core
  Qt6::Gui
  Qt6::GuiPrivate
  Qt6::Quick
  Qt6::Multimedia
)

target_compile_definitions(XMS PRIVATE QT_NO_KEYWORDS)
